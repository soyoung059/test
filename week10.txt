git 명령어

git branch "서브 브랜치 이름" (서브 브랜치 생성)
git switch "브랜치 이름" (브랜치 이동)
git push -u origin "서브 브랜치 이름" (메인 브랜치가 아니므로 1회 필요)

git merge "서브 브랜치 이름" (메인 브랜치로 이동 후 서브 브랜치를 병합, 충돌이 날 경우 해결 필요)
git branch -D "서브 브랜치 이름" (병합 완료 후 불필요한 서브 브랜치 로컬 리포지토리에서 제거)
git push origin -d "서브 브랜치 이름" (병합 완료 후 불필요한 서브 브랜치 원격 리포지토리에서 제거)
